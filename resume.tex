% Name command
\newcommand{\namesection}[3]{
\centering{
\fontsize{40pt}{60pt}

\fontspec[Path = fonts/lato/]{Lato-Lig}\selectfont #2

\color{headings}
\fontspec[Path = fonts/raleway/]{Raleway-Medium}\fontsize{11pt}{14pt}\selectfont #3}

\vspace{-15pt}
}
